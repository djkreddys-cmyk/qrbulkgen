# QRBulkGen Data Model Draft - March 13, 2026

## users
```sql
CREATE TABLE users (
  id UUID PRIMARY KEY,
  email VARCHAR(255) NOT NULL UNIQUE,
  password_hash TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

## qr_jobs
```sql
CREATE TABLE qr_jobs (
  id UUID PRIMARY KEY,
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  job_type VARCHAR(16) NOT NULL CHECK (job_type IN ('single', 'bulk')),
  status VARCHAR(16) NOT NULL CHECK (status IN ('queued', 'processing', 'completed', 'failed')),
  source_file_name TEXT,
  source_file_path TEXT,
  total_count INTEGER NOT NULL DEFAULT 0,
  success_count INTEGER NOT NULL DEFAULT 0,
  failure_count INTEGER NOT NULL DEFAULT 0,
  qr_content TEXT,
  qr_size INTEGER NOT NULL,
  foreground_color VARCHAR(32) NOT NULL,
  background_color VARCHAR(32) NOT NULL,
  qr_margin INTEGER NOT NULL,
  output_format VARCHAR(8) NOT NULL CHECK (output_format IN ('png', 'svg')),
  error_correction_level VARCHAR(1) NOT NULL CHECK (error_correction_level IN ('L', 'M', 'Q', 'H')),
  filename_prefix VARCHAR(120),
  error_message TEXT,
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

## qr_job_items
```sql
CREATE TABLE qr_job_items (
  id UUID PRIMARY KEY,
  job_id UUID NOT NULL REFERENCES qr_jobs(id) ON DELETE CASCADE,
  row_index INTEGER NOT NULL,
  content TEXT NOT NULL,
  status VARCHAR(16) NOT NULL CHECK (status IN ('queued', 'processing', 'completed', 'failed')),
  output_file_name TEXT,
  output_path TEXT,
  error_message TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

## job_artifacts
```sql
CREATE TABLE job_artifacts (
  id UUID PRIMARY KEY,
  job_id UUID NOT NULL REFERENCES qr_jobs(id) ON DELETE CASCADE,
  artifact_type VARCHAR(16) NOT NULL CHECK (artifact_type IN ('single-image', 'zip')),
  file_name TEXT NOT NULL,
  file_path TEXT NOT NULL,
  mime_type VARCHAR(100) NOT NULL,
  file_size_bytes BIGINT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

## analytics_events
```sql
CREATE TABLE analytics_events (
  id UUID PRIMARY KEY,
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  job_id UUID REFERENCES qr_jobs(id) ON DELETE SET NULL,
  event_type VARCHAR(32) NOT NULL,
  event_value INTEGER,
  metadata JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

## Indexes
```sql
CREATE INDEX idx_qr_jobs_user_created_at ON qr_jobs(user_id, created_at DESC);
CREATE INDEX idx_qr_jobs_status ON qr_jobs(status);
CREATE INDEX idx_qr_job_items_job_id ON qr_job_items(job_id);
CREATE INDEX idx_job_artifacts_job_id ON job_artifacts(job_id);
CREATE INDEX idx_analytics_events_user_created_at ON analytics_events(user_id, created_at DESC);
```

## Notes
- `qr_jobs` stores shared QR customization settings so single and bulk flows use the same shape.
- `qr_content` is populated for single jobs.
- `source_file_*` fields are populated for bulk jobs.
- `analytics_events` supports dashboard summary without blocking future aggregation changes.
